\textbf{Question:}
If \( \frac{a^n +b^n}{a^{n-1} + b^{n-1}}\)is A.M between a and b, then find value of n.
\break
\textbf{Solution:}
\begin{table}[h!]
    % Change address in github
    \input{table.tex}
    \caption{Parameters}
    \label{tab:12_13_3_13}
\end{table}\\
A.M between any two numbers a and b is average of those  numbers.\\
 We represent $a$, $b$, A.M of those two in an A.P\\

\begin{align}
    x(0)&=a\\
    x(1)&=A.M=\frac{x(0)+x(2)}{2}\label{1} \\
    x(2)&=b
\end{align}
From \eqref{1}
\begin{align}
    \frac{x(0)^n +x(2)^n}{x(0)^{n-1} + x(2)^{n-1}}&= \frac{x(0)+x(2)}{2}  \\
    \Rightarrow x(0)^n+x(2)^n&=x(2)x(0)^{n-1}+x(0)x(2)^{n-1} \\
    \Rightarrow x(0)^{n-1}(x(0)-x(2))&=x(2)^{n-1}(x(0)-x(2))\label{2}
 \end{align}


 From \eqref{2}-
\begin{align}
	n
	\begin{cases}
		=1  &\text{if } x(0)\neq x(2)\\
		\in R &\text{if } x(0)=x(2)
	\end{cases}
\end{align}
